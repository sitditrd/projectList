라벨 회전하여 출력
labelRotation : 90도로 hAxis축 & vAxis축 label을 회전 시킵니다.
<verticalAxisRenderers>
	<Axis2DRenderer axis="{hAxis}" labelRotation="90"/>
</verticalAxisRenderers>
<verticalAxisRenderers>
	<Axis2DRenderer axis="{vAxis}" labelRotation="90" tickPlacement="none"/>
</verticalAxisRenderers>
